  5  '''This examples shows how to submit a single BFAST  
  6     (bfast.sourceforge.net) genmoe matching job 
  7     to a remote PBS cluster via pbs+ssh. This example  
  8     is the same as pbs_via_ssh_simple_job.py, except 
  9     that it uses a differnet executable.  
 10     
 11     The example assuems that the BFAST executable and  
 12     data are present on the target system.  
 13   
 14     If something doesn't work as expected, try to set  
 15     SAGA_VERBOSE=3 in your environment before you run the 
 16     script in order to get some debug output.

 32      try: 
 33          # set up a security context (optional) 
 34          # if no security context is defined, the PBS 
 35          # plugin will pick up the default set of ssh  
 36          # credentials of the user, i.e., ~/.ssh/id_rsa 
 37          ctx = saga.Context() 
 38          ctx.type = saga.Context.SSH 
 39          ctx.userid  = 'oweidner' # like 'ssh username@host ...' 
 40   
 41          # create a job service for Futuregrid's 'india' PBS cluster 
 42          # and attach the SSH security context to it 
 43          js = saga.job.Service("pbs+ssh://india.futuregrid.org") 
 44          js.session.contexts.append(ctx) 
 45   
 46          # describe our job 
 47          jd = saga.job.Description() 
 48          # resource requirements 
 49          jd.wall_time_limit  = "0:05:00" 
 50          jd.total_cpu_count = 1      
 51          # environment, executable & arguments 
 52          jd.environment       = {'BFAST_DIR':'/N/u/oweidner/bfast'} 
 53          jd.working_directory = '/N/u/oweidner/bfast/tmp/'      
 54          jd.executable        = '$BFAST_DIR/bin/bfast' 
 55          jd.arguments         = ['match', '-A 1', 
 56                                  '-r $BFAST_DIR/data/small/reads_5K/reads.10.fastq', 
 57                                  '-f $BFAST_DIR/data/small/reference/hg_2122.fa'] 
 58          # output options 
 59          localtime = time.time() 
 60          jd.output = "bfast_match_%s.stdout" % localtime 
 61          jd.error  = "bfast_match_%s.stderr" % localtime 
 62   
 63          # create the job (state: New) 
 64          myjob = js.create_job(jd) 
 65   
 66          print "Job ID    : %s" % (myjob.jobid) 
 67          print "Job State : %s" % (myjob.get_state()) 
 68   
 69          print "\n...starting job...\n" 
 70          # run the job (submit the job to PBS) 
 71          myjob.run() 
 72   
 73          print "Job ID    : %s" % (myjob.jobid) 
 74          print "Job State : %s" % (myjob.get_state()) 
 75   
 76          print "\n...waiting for job...\n" 
 77          # wait for the job to either finish or fail 
 78          myjob.wait() 
 79   
 80          print "Job State : %s" % (myjob.get_state()) 
 81          print "Exitcode  : %s" % (myjob.exitcode) 
 82   
 83      except saga.Exception, ex: 
 84          print "Oh, snap! An error occured: %s" % (str(ex))
